create table "ROOT".ALUNO
(
	ID INTEGER GENERATED ALWAYS AS IDENTITY(START WITH 1, INCREMENT BY 1) not null primary key,
	NOME VARCHAR(255) not null,	
	DOCUMENTO VARCHAR(255) not null,
	DTNASCIMENTO DATE not null,
	ENDERECO VARCHAR(255) not null,
	CIDADE VARCHAR(255) not null,
	ESTADO VARCHAR(255) not null,
	EMAIL VARCHAR(255) not null,
	TELEFONE VARCHAR(255) not null,
	CELULAR VARCHAR(255),
	CADASTRADOEM TIMESTAMP not null,
	CADASTRADOPOR VARCHAR(255) not null,
	ENABLED BOOLEAN default false not null
);

create table "ROOT".FUNCIONARIO
(
	ID INTEGER GENERATED ALWAYS AS IDENTITY(START WITH 1, INCREMENT BY 1) not null primary key,
	NOME VARCHAR(255) not null,	
	DOCUMENTO VARCHAR(255) not null,
	DTNASCIMENTO DATE not null,
	ENDERECO VARCHAR(255) not null,
	CIDADE VARCHAR(255) not null,
	ESTADO VARCHAR(255) not null,
	EMAIL VARCHAR(255) not null,
	CARGO VARCHAR(255) not null,	
	USERNAME VARCHAR(255) not null,
	HASHSENHA VARCHAR(255) not null,	
	CADASTRADOEM TIMESTAMP not null,
	CADASTRADOPOR VARCHAR(255) not null,
	ENABLED BOOLEAN default false not null
);

create table "ROOT".CURSO
(
	ID INTEGER GENERATED ALWAYS AS IDENTITY(START WITH 1, INCREMENT BY 1)  not null primary key,
	NOME VARCHAR(255) not null,
	CATEGORIA VARCHAR(255) not null,
	DESCRICAO VARCHAR(255),
	VALOR DECIMAL(8,2) not null,
	CADASTRADOEM TIMESTAMP not null,
	CADASTRADOPOR VARCHAR(255) not null,
	ENABLED BOOLEAN default false not null
);

create table "ROOT".UNIDADE
(
	ID INTEGER GENERATED ALWAYS AS IDENTITY(START WITH 1, INCREMENT BY 1) not null primary key,
	NOME VARCHAR(255) not null,
	ENDERECO VARCHAR(255) not null,
	CIDADE VARCHAR(255) not null,
	ESTADO VARCHAR(255) not null,
	EMAIL VARCHAR(255) not null,
	TELEFONE1 VARCHAR(255) not null,
	TELEFONE2 VARCHAR(255) not null,	
	CADASTRADOEM TIMESTAMP not null,
	CADASTRADOPOR VARCHAR(255) not null,
	ENABLED BOOLEAN default false not null	
);

create table "ROOT".MATRICULA
(
	IDMATRICULA INTEGER GENERATED ALWAYS AS IDENTITY(START WITH 1, INCREMENT BY 1)  not null primary key,
	IDALUNO INTEGER not null,
	IDUNIDADE INTEGER not null,
	IDCURSO INTEGER not null,	
	DTINICIO DATE not null,
	PAGAMENTO VARCHAR(255) not null,	
	CADASTRADOEM TIMESTAMP not null,
	CADASTRADOPOR VARCHAR(255) not null,
	ENABLED BOOLEAN default false not null,
	FOREIGN KEY (IDALUNO) REFERENCES aluno(id),
	FOREIGN KEY (IDUNIDADE) REFERENCES unidade(id),
	FOREIGN KEY (IDCURSO) REFERENCES curso(id)
);
